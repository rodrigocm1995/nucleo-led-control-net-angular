Firmware de la Tarjeta (STM32Cube HAL)

/* USER CODE BEGIN Header */
/**
  * @file           : main.c
  * @brief          : Control de LED por comandos serie UART2 (PA5)
  ******************************************************************************
  */
#include "main.h"
#include "stm32f3xx_hal.h"
#include <stdio.h>
#include <string.h>

UART_HandleTypeDef huart2;
uint8_t userInput = 0;
char tx_data[50];

int main(void)
{
  HAL_Init();
  SystemClock_Config();
  MX_GPIO_Init();
  MX_USART2_UART_Init();

  while (1)
  {
    // Leer carácter del puerto serie por polling
    if (HAL_UART_Receive(&huart2, &userInput, 1, 100) == HAL_OK)
    {
      switch(userInput)
      {
        case '0': // Apagar LED
          HAL_GPIO_WritePin(GPIOA, GPIO_PIN_5, GPIO_PIN_RESET);
          sprintf(tx_data, "LED Apagado\r\n");
          HAL_UART_Transmit(&huart2, (uint8_t *)tx_data, strlen(tx_data), 100);
          break;
        case '1': // Enciende LED
          HAL_GPIO_WritePin(GPIOA, GPIO_PIN_5, GPIO_PIN_SET);
          sprintf(tx_data, "LED Encendido\r\n");
          HAL_UART_Transmit(&huart2, (uint8_t *)tx_data, strlen(tx_data), 100);
          break;
        case '?': // Consultar Estado
          if ((GPIOA->ODR & GPIO_PIN_5) != 0) {
            sprintf(tx_data, "LED Encendido\r\n");
          } else {
            sprintf(tx_data, "LED Apagado\r\n");
          }
          HAL_UART_Transmit(&huart2, (uint8_t *)tx_data, strlen(tx_data), 100);
          break;
      }
    }
    HAL_Delay(1);
  }
}
